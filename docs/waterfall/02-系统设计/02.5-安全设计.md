## 1. 安全目标与威胁模型

### 1.1 安全目标

| 目标 | 优先级 | 说明 |
|------|--------|------|
| G1 | P0 | 创始人数据主权：所有数据本地存储，不上传云端 |
| G2 | P0 | 操作可追溯：所有Agent操作记录审计日志，不可篡改 |
| G3 | P0 | 权限最小化：Agent只能访问授权的工具和数据 |
| G4 | P1 | API Key安全：云端API Key加密存储，不明文暴露 |
| G5 | P1 | 代码执行安全：Agent生成的代码在沙箱中执行 |
| G6 | P2 | 成本防滥用：Token消耗熔断，防止意外高额账单 |

### 1.2 威胁模型

| 威胁 | 风险等级 | 影响 | 缓解措施 |
|------|---------|------|---------|
| Agent误操作删除文件 | 高 | 数据丢失 | 文件操作权限分级 + 审计日志 |
| Agent生成恶意代码 | 中 | 系统受损 | 沙箱执行 + 审计Agent审查 |
| API Key泄露 | 中 | 云服务被盗用 | 加密存储 + 环境变量隔离 |
| 本地数据库被篡改 | 中 | 审计失效 | 审计日志append-only + 定期备份 |
| 内存不足导致崩溃 | 中 | 任务中断 | 内存监控 + 任务优先级调度 |
| 创始人误操作 | 低 | 配置错误 | 操作确认对话框 + 撤销机制 |

### 1.3 信任边界

```
┌─────────────────────────────────────────────────────────────┐
│                    信任边界: 创始人(从配置读取，默认"老板")                        │
│  ┌─────────────────────────────────────────────────────┐   │
│  │  信任边界: S-O-P 系统进程                              │   │
│  │  ┌─────────────────────────────────────────────┐   │   │
│  │  │  信任边界: Agent 沙箱                          │   │   │
│  │  │  ┌─────────────────────────────────────────┐ │   │   │
│  │  │  │  不信任: LLM 输出                         │ │   │   │
│  │  │  │  ┌─────────────────────────────────────┐│ │   │   │
│  │  │  │  │  不信任: 云端 API (DeepSeek等)        ││ │   │   │
│  │  │  │  └─────────────────────────────────────┘│ │   │   │
│  │  │  └─────────────────────────────────────────┘ │   │   │
│  │  └─────────────────────────────────────────────┘   │   │
│  └─────────────────────────────────────────────────────┘   │
└─────────────────────────────────────────────────────────────┘
```

**关键原则**:
- 创始人完全信任系统进程（因为是自己部署的）
- 系统进程不信任Agent生成的代码（必须沙箱执行）
- 系统进程不信任LLM输出（必须审计Agent审查）
- 系统进程不信任云端API（API Key加密，请求签名验证）

## 2. 权限模型

### 2.1 权限分级

```python
from enum import Enum

class PermissionLevel(Enum):
    """权限级别"""
    P0 = "founder"      # 创始人: 所有权限
    P1 = "read"         # 读取: 文件读取、数据库查询、日志查看
    P2 = "write"        # 写入: 文件写入、数据库修改、Git提交
    P3 = "execute"      # 执行: 代码执行、命令执行、系统调用
    P4 = "admin"        # 管理: 配置修改、Agent启停、数据删除

class Permission:
    """权限定义"""

    # 权限矩阵: 操作 -> 所需权限级别
    OPERATION_PERMISSIONS = {
        # 文件操作
        "file_read": PermissionLevel.P1,
        "file_write": PermissionLevel.P2,
        "file_delete": PermissionLevel.P4,

        # 数据库操作
        "db_query": PermissionLevel.P1,
        "db_insert": PermissionLevel.P2,
        "db_update": PermissionLevel.P2,
        "db_delete": PermissionLevel.P4,

        # 代码执行
        "code_execute": PermissionLevel.P3,
        "code_execute_unrestricted": PermissionLevel.P0,  # 创始人专用

        # Git操作
        "git_status": PermissionLevel.P1,
        "git_commit": PermissionLevel.P2,
        "git_push": PermissionLevel.P4,

        # 系统操作
        "system_shutdown": PermissionLevel.P0,
        "system_restart": PermissionLevel.P0,
        "config_read": PermissionLevel.P1,
        "config_write": PermissionLevel.P4,

        # Agent管理
        "agent_start": PermissionLevel.P4,
        "agent_stop": PermissionLevel.P4,
        "agent_config": PermissionLevel.P4,
    }

    # Agent默认权限
    AGENT_DEFAULT_PERMISSIONS = {
        "agent_main": [PermissionLevel.P1, PermissionLevel.P2],
        "agent_audit": [PermissionLevel.P1],
        "agent_dev": [PermissionLevel.P1, PermissionLevel.P2, PermissionLevel.P3],
    }

    # 创始人权限
    FOUNDER_PERMISSIONS = [PermissionLevel.P0, PermissionLevel.P1,
                           PermissionLevel.P2, PermissionLevel.P3, PermissionLevel.P4]
```

### 2.2 权限检查流程

```python
class PermissionChecker:
    """权限检查器"""

    def __init__(self, db: Database):
        self.db = db

    def check_permission(
        self,
        subject: str,           # 主体: agent_id 或 "user"(创始人)
        operation: str,         # 操作
        target: Optional[str] = None,  # 目标对象
        context: Optional[dict] = None   # 上下文
    ) -> PermissionResult:
        """
        检查主体是否有权限执行操作

        流程:
        1. 创始人(user) -> 直接通过
        2. Agent -> 检查Agent权限级别
        3. 检查操作所需权限
        4. 检查目标对象特殊限制
        5. 记录权限检查日志
        """

        # 1. 创始人拥有所有权限
        if subject == "user":
            return PermissionResult(
                allowed=True,
                reason="创始人拥有所有权限",
                level=PermissionLevel.P0
            )

        # 2. 获取Agent权限
        agent_perms = self._get_agent_permissions(subject)

        # 3. 获取操作所需权限
        required_perm = Permission.OPERATION_PERMISSIONS.get(operation)
        if required_perm is None:
            # 未知操作，默认拒绝
            return PermissionResult(
                allowed=False,
                reason=f"未知操作: {operation}",
                level=None
            )

        # 4. 检查权限是否满足
        if required_perm not in agent_perms:
            return PermissionResult(
                allowed=False,
                reason=f"Agent '{subject}' 无权限执行 '{operation}' (需要 {required_perm.value}, 拥有 {', '.join(p.value for p in agent_perms)})",
                level=required_perm
            )

        # 5. 检查目标对象特殊限制
        if target and not self._check_target_restrictions(subject, operation, target):
            return PermissionResult(
                allowed=False,
                reason=f"Agent '{subject}' 无权访问目标 '{target}'",
                level=required_perm
            )

        # 6. 记录权限检查日志
        self._log_permission_check(subject, operation, target, True)

        return PermissionResult(
            allowed=True,
            reason="权限检查通过",
            level=required_perm
        )

    def _get_agent_permissions(self, agent_id: str) -> list[PermissionLevel]:
        """获取Agent权限列表"""
        # 从数据库查询Agent配置
        agent = self.db.query(
            "SELECT permissions FROM agents WHERE agent_id = ?",
            [agent_id]
        )

        if agent:
            # 解析权限字符串
            perms_str = agent[0]['permissions']
            return [PermissionLevel[p.strip()] for p in perms_str.split(',')]

        # 默认权限
        return Permission.AGENT_DEFAULT_PERMISSIONS.get(agent_id, [PermissionLevel.P1])

    def _check_target_restrictions(
        self,
        agent_id: str,
        operation: str,
        target: str
    ) -> bool:
        """检查目标对象特殊限制"""

        # 禁止访问的目录/文件
        forbidden_paths = [
            '.env',                    # 环境变量文件
            'data/solo_ops.db',        # 主数据库
            'data/solo_ops.db.backup', # 数据库备份
            'data/chromadb',           # 向量数据库
            '.git/config',            # Git配置(可能含凭证)
            '__pycache__',             # 缓存
        ]

        # 检查目标是否在禁止列表中
        for forbidden in forbidden_paths:
            if forbidden in target:
                return False

        # 检查路径是否在项目目录内
        import os
        base_path = os.path.abspath('.')
        target_path = os.path.abspath(target)
        if not target_path.startswith(base_path):
            return False

        return True

    def _log_permission_check(
        self,
        subject: str,
        operation: str,
        target: Optional[str],
        allowed: bool
    ):
        """记录权限检查日志"""
        self.db.execute("""
            INSERT INTO audit_log (timestamp, agent_id, operation_type, target_type, target_id, params, result)
            VALUES (datetime('now'), ?, 'permission_check', ?, ?, ?, ?)
        """, [
            subject,
            'operation',
            target or operation,
            json.dumps({"operation": operation, "target": target}),
            "ALLOWED" if allowed else "DENIED"
        ])

@dataclass
class PermissionResult:
    """权限检查结果"""
    allowed: bool
    reason: str
    level: Optional[PermissionLevel]
```

### 2.3 工具调用权限控制

```python
class SecureToolKit(ToolKit):
    """带权限控制的工具调用框架"""

    def __init__(self, db: Database, permission_checker: PermissionChecker):
        super().__init__(db)
        self.permission_checker = permission_checker

    def call_tool(
        self,
        tool_name: str,
        agent_id: str,
        params: dict,
        task_id: Optional[str] = None
    ) -> dict:
        """
        带权限检查的工具调用

        流程:
        1. 检查Agent是否有权限调用此工具
        2. 检查参数中的目标路径是否允许访问
        3. 执行工具
        4. 记录审计日志
        """

        # 1. 权限检查
        # 将tool_name映射到operation
        operation_map = {
            "file_read": "file_read",
            "file_write": "file_write",
            "db_query": "db_query",
            "code_execute": "code_execute",
            "git_commit": "git_commit",
        }
        operation = operation_map.get(tool_name, tool_name)

        # 获取目标路径(如果有)
        target = params.get("path") or params.get("file_path") or params.get("sql")

        permission_result = self.permission_checker.check_permission(
            subject=agent_id,
            operation=operation,
            target=target
        )

        if not permission_result.allowed:
            # 权限拒绝，记录并返回错误
            self.db.execute("""
                INSERT INTO audit_log (timestamp, agent_id, operation_type, target_type, target_id, params, result)
                VALUES (datetime('now'), ?, 'tool_call', ?, ?, ?, ?)
            """, [
                agent_id,
                'tool',
                tool_name,
                json.dumps(params),
                f"DENIED: {permission_result.reason}"
            ])

            return {
                "success": False,
                "error": permission_result.reason,
                "permission_denied": True
            }

        # 2. 执行工具(调用父类方法)
        return super().call_tool(tool_name, agent_id, params, task_id)
```

## 3. 数据安全

### 3.1 数据分类

| 分类 | 数据 | 敏感度 | 保护措施 |
|------|------|--------|---------|
| 核心机密 | API Key、数据库密码 | 最高 | 加密存储，内存中解密 |
| 敏感数据 | 审计日志、任务记录 | 高 | 本地存储，定期备份 |
| 业务数据 | SOP模板、Agent配置 | 中 | 本地存储，版本控制 |
| 公开数据 | 代码文件、文档 | 低 | 常规文件权限 |

### 3.2 API Key 加密存储

```python
from cryptography.fernet import Fernet
import os
import base64

class SecureConfig:
    """安全配置管理器 —— API Key等敏感数据的唯一存储位置

    设计原则:
    - SQLite config表仅存储非敏感配置（预算、阈值、主题等）
    - 所有敏感数据（API Key、密码等）必须通过此类存储
    - 使用Fernet对称加密，密钥存储在data/.secret_key
    """

    def __init__(self, config_path: str = "data/config.json"):
        self.config_path = config_path
        self._key = self._get_or_create_key()
        self._cipher = Fernet(self._key)
        self._config = self._load_config()

    def _get_or_create_key(self) -> bytes:
        """获取或创建加密密钥"""
        key_path = "data/.secret_key"

        if os.path.exists(key_path):
            with open(key_path, 'rb') as f:
                return base64.urlsafe_b64decode(f.read())

        # 生成新密钥
        key = Fernet.generate_key()
        os.makedirs(os.path.dirname(key_path), exist_ok=True)
        with open(key_path, 'wb') as f:
            f.write(base64.urlsafe_b64encode(key))

        # 设置文件权限: 仅所有者可读写
        os.chmod(key_path, 0o600)

        return key

    def _load_config(self) -> dict:
        """加载配置"""
        if not os.path.exists(self.config_path):
            return {}

        import json
        with open(self.config_path, 'r') as f:
            return json.load(f)

    def get(self, key: str, default=None) -> any:
        """获取明文配置项"""
        return self._config.get(key, default)

    def get_encrypted(self, key: str) -> Optional[str]:
        """获取加密配置项并解密"""
        encrypted_value = self._config.get(key)
        if not encrypted_value:
            return None

        try:
            decrypted = self._cipher.decrypt(encrypted_value.encode())
            return decrypted.decode()
        except Exception as e:
            logger.error(f"解密配置项失败: {key}, {e}")
            return None

    def set(self, key: str, value: any, encrypt: bool = False):
        """设置配置项"""
        if encrypt:
            # 加密存储
            encrypted = self._cipher.encrypt(value.encode()).decode()
            self._config[key] = encrypted
        else:
            self._config[key] = value

        self._save_config()

    def _save_config(self):
        """保存配置"""
        import json
        os.makedirs(os.path.dirname(self.config_path), exist_ok=True)
        with open(self.config_path, 'w') as f:
            json.dump(self._config, f, indent=2, ensure_ascii=False)

        # 设置文件权限
        os.chmod(self.config_path, 0o600)

# 使用示例
config = SecureConfig()

# 存储API Key(加密)
config.set("api_key", "sk-xxxxxxxxxxxxxxxx", encrypt=True)

# 读取API Key(自动解密)
api_key = config.get_encrypted("api_key")

# 存储普通配置(不加密)
config.set("ollama_host", "http://localhost:11434")
```

### 3.3 数据库安全

```python
class SecureDatabase(Database):
    """带安全控制的数据库访问层"""

    def __init__(self, db_path: str = "data/solo_ops.db"):
        super().__init__(db_path)
        self._init_security()

    def _init_security(self):
        """初始化安全设置"""
        conn = self._get_connection()

        # 启用WAL模式(更好的并发和恢复)
        conn.execute("PRAGMA journal_mode = WAL")

        # 启用外键约束
        conn.execute("PRAGMA foreign_keys = ON")

        # 设置同步模式(平衡性能和安全性)
        conn.execute("PRAGMA synchronous = NORMAL")

    def execute(self, sql: str, params: list = None) -> int:
        """执行写入(带安全检查)"""

        # 安全检查1: 禁止DROP TABLE
        if "DROP TABLE" in sql.upper():
            raise SecurityError("禁止删除表操作")

        # 安全检查2: 禁止DELETE FROM audit_log(审计日志append-only)
        if "DELETE FROM audit_log" in sql.upper() or "DELETE FROM cost_log" in sql.upper():
            raise SecurityError("审计日志和成本日志禁止删除")

        # 安全检查3: 参数化查询(防止SQL注入)
        # 父类已实现参数化查询

        return super().execute(sql, params)

    def backup(self, backup_path: Optional[str] = None) -> str:
        """创建数据库备份"""
        if backup_path is None:
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            backup_path = f"data/backups/solo_ops_{timestamp}.db"

        os.makedirs(os.path.dirname(backup_path), exist_ok=True)

        conn = self._get_connection()
        backup_conn = sqlite3.connect(backup_path)
        conn.backup(backup_conn)
        backup_conn.close()

        # 设置备份文件权限
        os.chmod(backup_path, 0o600)

        return backup_path

class SecurityError(Exception):
    """安全错误"""
    pass
```

## 4. 审计与合规

### 4.1 审计日志设计

审计日志是系统的**安全核心**，必须满足以下要求：

| 要求 | 实现 | 验证方式 |
|------|------|---------|
| 不可删除 | SQLite触发器阻止DELETE | 尝试删除会报错 |
| 不可修改 | 无UPDATE接口 | 代码审查确认 |
| 完整性 | 时间戳 + 顺序ID | 检查ID连续性 |
| 可追溯 | 记录Agent、操作、参数、结果 | 查询可还原完整操作链 |
| 防篡改 | 定期备份 + 哈希校验 | 备份文件对比 |

### 4.2 审计日志表结构

```sql
-- 审计日志表(append-only)
CREATE TABLE audit_log (
    log_id INTEGER PRIMARY KEY AUTOINCREMENT,  -- 自增ID，保证顺序
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,  -- 操作时间
    agent_id TEXT,                               -- 操作主体
    operation_type TEXT NOT NULL,                -- 操作类型
    target_type TEXT,                            -- 目标类型
    target_id TEXT,                              -- 目标ID
    params TEXT,                                 -- 参数(JSON)
    result TEXT,                                 -- 结果
    duration_ms INTEGER,                         -- 耗时
    cost REAL,                                   -- 成本
    tokens INTEGER,                              -- Token数
    ip_address TEXT,                             -- IP地址(如适用)
    session_id TEXT                              -- 会话ID
);

-- 防止删除的触发器
CREATE TRIGGER prevent_audit_log_delete
BEFORE DELETE ON audit_log
BEGIN
    SELECT RAISE(ABORT, '审计日志禁止删除');
END;

-- 防止修改的触发器
CREATE TRIGGER prevent_audit_log_update
BEFORE UPDATE ON audit_log
BEGIN
    SELECT RAISE(ABORT, '审计日志禁止修改');
END;

-- 索引(加速查询)
CREATE INDEX idx_audit_agent ON audit_log(agent_id);
CREATE INDEX idx_audit_operation ON audit_log(operation_type);
CREATE INDEX idx_audit_timestamp ON audit_log(timestamp);
CREATE INDEX idx_audit_target ON audit_log(target_type, target_id);
```

### 4.3 审计日志查询接口

```python
class AuditLogQuery:
    """审计日志查询"""

    def __init__(self, db: SecureDatabase):
        self.db = db

    def get_agent_operations(
        self,
        agent_id: str,
        start_time: Optional[datetime] = None,
        end_time: Optional[datetime] = None,
        operation_type: Optional[str] = None
    ) -> list[dict]:
        """查询指定Agent的操作记录"""
        sql = "SELECT * FROM audit_log WHERE agent_id = ?"
        params = [agent_id]

        if start_time:
            sql += " AND timestamp >= ?"
            params.append(start_time.isoformat())
        if end_time:
            sql += " AND timestamp <= ?"
            params.append(end_time.isoformat())
        if operation_type:
            sql += " AND operation_type = ?"
            params.append(operation_type)

        sql += " ORDER BY timestamp DESC"

        return self.db.query(sql, params)

    def get_task_trace(self, task_id: str) -> list[dict]:
        """获取任务的完整操作链"""
        # 查询所有与此任务相关的操作
        sql = """
            SELECT * FROM audit_log
            WHERE params LIKE ?
               OR target_id = ?
               OR (target_type = 'task' AND target_id = ?)
            ORDER BY timestamp ASC
        """
        return self.db.query(sql, [f'%"task_id": "{task_id}"%', task_id, task_id])

    def get_cost_summary(
        self,
        start_time: Optional[datetime] = None,
        end_time: Optional[datetime] = None
    ) -> dict:
        """获取成本汇总"""
        sql = """
            SELECT
                agent_id,
                COUNT(*) as operation_count,
                SUM(cost) as total_cost,
                SUM(tokens) as total_tokens,
                AVG(duration_ms) as avg_duration
            FROM audit_log
            WHERE cost > 0
        """
        params = []

        if start_time:
            sql += " AND timestamp >= ?"
            params.append(start_time.isoformat())
        if end_time:
            sql += " AND timestamp <= ?"
            params.append(end_time.isoformat())

        sql += " GROUP BY agent_id"

        return self.db.query(sql, params)

    def get_integrity_check(self) -> dict:
        """审计日志完整性检查"""
        # 检查ID连续性
        result = self.db.query("""
            SELECT
                COUNT(*) as total_records,
                MIN(log_id) as min_id,
                MAX(log_id) as max_id,
                MAX(log_id) - MIN(log_id) + 1 as expected_count
            FROM audit_log
        """)

        if result:
            r = result[0]
            return {
                "total_records": r['total_records'],
                "min_id": r['min_id'],
                "max_id": r['max_id'],
                "expected_count": r['expected_count'],
                "is_continuous": r['total_records'] == r['expected_count'],
                "integrity_status": "OK" if r['total_records'] == r['expected_count'] else "BROKEN"
            }

        return {"integrity_status": "EMPTY"}
```

```python
import subprocess
import tempfile

class CodeSandbox:
    """代码执行沙箱"""

    def __init__(self, timeout: int = 30, max_memory_mb: int = 512):
        self.timeout = timeout
        self.max_memory_mb = max_memory_mb
        self.allowed_modules = [
            'os', 'sys', 'json', 're', 'datetime', 'math', 'random',
            'collections', 'itertools', 'functools', 'typing',
            'sqlite3', 'pathlib', 'hashlib', 'base64', 'urllib.parse',
            'requests',  # 网络请求(限制)
        ]
        self.forbidden_patterns = [
            'import subprocess',
            'os.system',
            'os.popen',
            'eval(',
            'exec(',
            '__import__',
            'open("/etc',
            'open("C:\\Windows',
        ]

    def execute(self, code: str, context: dict = None) -> dict:
        """
        在沙箱中执行代码

        安全措施:
        1. 语法检查
        2. 禁止模式检查
        3. 子进程隔离
        4. 超时控制
        5. 内存限制
        6. 文件系统限制
        """

        # 1. 语法检查
        try:
            compile(code, '<sandbox>', 'exec')
        except SyntaxError as e:
            return {
                "success": False,
                "error": f"语法错误: {e}",
                "stdout": "",
                "stderr": ""
            }

        # 2. 禁止模式检查
        for pattern in self.forbidden_patterns:
            if pattern in code:
                return {
                    "success": False,
                    "error": f"安全错误: 禁止使用的模式 '{pattern}'",
                    "stdout": "",
                    "stderr": ""
                }

        # 3. 创建临时文件
        # 注意: Windows不支持Unix的resource模块进行内存限制
        # 安全边界依赖: (1)语法检查 (2)禁止模式检查 (3)subprocess超时 (4)工作目录限制
        # V1.1+可考虑引入更严格的沙箱方案(docker/RestrictedPython)
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            # 写入安全包装代码（跨平台兼容，不使用resource模块）
            wrapper = f'''
import sys
import os

# 限制文件系统访问（工作目录限制）
os.chdir('{os.getcwd()}')

# 执行用户代码
{code}
'''
            f.write(wrapper)
            temp_path = f.name

        try:
            # 4. 子进程执行
            result = subprocess.run(
                ['python', temp_path],
                capture_output=True,
                text=True,
                timeout=self.timeout,
                cwd=os.getcwd(),
                # 限制环境变量
                env={**os.environ, 'PYTHONPATH': os.getcwd()}
            )

            return {
                "success": result.returncode == 0,
                "stdout": result.stdout,
                "stderr": result.stderr,
                "returncode": result.returncode
            }

        except subprocess.TimeoutExpired:
            return {
                "success": False,
                "error": f"执行超时(>{self.timeout}秒)",
                "stdout": "",
                "stderr": ""
            }
        except Exception as e:
            return {
                "success": False,
                "error": f"执行错误: {str(e)}",
                "stdout": "",
                "stderr": ""
            }
        finally:
            # 清理临时文件
            try:
                os.unlink(temp_path)
            except:
                pass
```

```python
class ResourceLimiter:
    """资源限制器"""

    def __init__(self):
        self.limits = {
            "max_memory_mb": 1024,        # 最大内存1GB
            "max_disk_mb": 1024,          # 最大磁盘1GB
            "max_cpu_percent": 80,         # 最大CPU 80%
            "max_concurrent_tasks": 3,     # 最大并发任务3个
            "max_llm_calls_per_minute": 10,  # 每分钟最大LLM调用10次
        }
        self.current_usage = {
            "memory_mb": 0,
            "disk_mb": 0,
            "cpu_percent": 0,
            "concurrent_tasks": 0,
            "llm_calls_last_minute": 0,
        }

    def check_resource(self, resource_type: str, required: float) -> bool:
        """检查资源是否足够"""
        limit = self.limits.get(resource_type)
        current = self.current_usage.get(resource_type, 0)

        if limit is None:
            return True

        return (current + required) <= limit

    def acquire_resource(self, resource_type: str, amount: float):
        """申请资源"""
        if not self.check_resource(resource_type, amount):
            raise ResourceError(f"资源不足: {resource_type} (需要 {amount}, 剩余 {self.limits[resource_type] - self.current_usage[resource_type]})")

        self.current_usage[resource_type] += amount

    def release_resource(self, resource_type: str, amount: float):
        """释放资源"""
        self.current_usage[resource_type] = max(0, self.current_usage[resource_type] - amount)

    def get_status(self) -> dict:
        """获取资源状态"""
        return {
            resource: {
                "limit": self.limits[resource],
                "current": self.current_usage[resource],
                "available": self.limits[resource] - self.current_usage[resource],
                "usage_percent": (self.current_usage[resource] / self.limits[resource] * 100) if self.limits[resource] > 0 else 0
            }
            for resource in self.limits.keys()
        }

class ResourceError(Exception):
    """资源错误"""
    pass
```

```python
        # 配置安全设置
        self.session.verify = True  # 验证SSL证书
        self.session.timeout = (5, 60)  # (连接超时, 读取超时)

    def call_api(self, endpoint: str, data: dict) -> dict:
        """
        安全调用云端API

        安全措施:
        1. HTTPS强制
        2. 请求签名
        3. 超时控制
        4. 响应验证
        5. 错误处理
        """

                json=data,
                timeout=self.session.timeout

            response.raise_for_status()

            # 5. 验证响应

            # 6. 清除敏感数据
            del api_key

        except requests.exceptions.SSLError:

        except requests.exceptions.Timeout:
```

```python
import shutil
from datetime import datetime, timedelta

class BackupManager:
    """备份管理器"""

    def __init__(self, db: SecureDatabase):
        self.db = db
        self.backup_dir = Path("data/backups")
        self.backup_dir.mkdir(parents=True, exist_ok=True)

    def create_backup(self, backup_type: str = "manual") -> str:
        """创建备份"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")

        # 备份数据库
        db_backup = self.backup_dir / f"solo_ops_{timestamp}.db"
        self.db.backup(str(db_backup))

        # 备份配置
        config_backup = self.backup_dir / f"config_{timestamp}.json"
        if Path("data/config.json").exists():
            shutil.copy("data/config.json", config_backup)

        # 备份向量库(每周或手动)
        if backup_type == "weekly" or backup_type == "manual":
            chroma_backup = self.backup_dir / f"chromadb_{timestamp}"
            if Path("data/chromadb").exists():
                shutil.copytree("data/chromadb", chroma_backup)

        # 记录备份日志
        self.db.execute("""
            INSERT INTO audit_log (timestamp, agent_id, operation_type, target_type, target_id, params, result)
            VALUES (datetime('now'), 'system', 'backup', 'system', ?, ?, 'SUCCESS')
        """, [
            f"backup_{timestamp}",
            json.dumps({"type": backup_type, "files": [str(db_backup), str(config_backup)]})
        ])

        return str(db_backup)

    def cleanup_old_backups(self, days: int = 7):
        """清理旧备份"""
        cutoff = datetime.now() - timedelta(days=days)

        for backup_file in self.backup_dir.glob("*.db"):
            # 从文件名解析日期
            try:
                file_date = datetime.strptime(backup_file.stem.split('_')[1], "%Y%m%d")
                if file_date < cutoff:
                    backup_file.unlink()
                    logger.info(f"删除旧备份: {backup_file}")
            except:
                pass

    def restore_backup(self, backup_path: str) -> bool:
        """从备份恢复"""
        backup_path = Path(backup_path)

        if not backup_path.exists():
            raise SecurityError(f"备份文件不存在: {backup_path}")

        # 创建当前数据备份(防止恢复失败)
        self.create_backup("pre_restore")

        # 恢复数据库
        db_path = Path("data/solo_ops.db")
        shutil.copy(backup_path, db_path)

        # 记录恢复操作
        self.db.execute("""
            INSERT INTO audit_log (timestamp, agent_id, operation_type, target_type, target_id, params, result)
            VALUES (datetime('now'), 'user', 'restore', 'system', ?, ?, 'SUCCESS')
        """, [
            str(backup_path),
            json.dumps({"restored_from": str(backup_path)})
        ])

        return True
```

## 8. 安全事件响应

### 8.1 安全事件分类

| 级别 | 事件 | 响应措施 |
|------|------|---------|
| 紧急 | 数据库被篡改 | 立即停止所有Agent，恢复备份，调查原因 |
| 紧急 | API Key泄露 | 立即更换API Key，检查账单，通知提供商 |
| 高 | Agent权限越界 | 暂停Agent，审查权限配置，修复漏洞 |
| 高 | 代码执行沙箱逃逸 | 停止系统，更新沙箱规则，审查代码 |
| 中 | 成本异常增长 | 触发熔断，审查任务，优化配置 |
| 低 | 配置错误 | 提示用户，提供修复建议 |

### 8.2 安全事件响应流程

```python
class SecurityIncidentResponse:
    """安全事件响应"""

    def __init__(self, db: SecureDatabase, scheduler: Scheduler):
        self.db = db
        self.scheduler = scheduler

    def handle_incident(self, incident_type: str, details: dict):
        """处理安全事件"""

        # 1. 记录事件
        self.db.execute("""
            INSERT INTO audit_log (timestamp, agent_id, operation_type, target_type, target_id, params, result)
            VALUES (datetime('now'), 'system', 'security_incident', ?, ?, ?, ?)
        """, [
            incident_type,
            details.get('target', 'system'),
            json.dumps(details),
            "DETECTED"
        ])

        # 2. 根据级别响应
        if incident_type in ['db_tampered', 'api_key_leaked', 'sandbox_escape']:
            self._handle_critical_incident(incident_type, details)
        elif incident_type in ['permission_violation', 'cost_anomaly']:
            self._handle_high_incident(incident_type, details)
        else:
            self._handle_low_incident(incident_type, details)

    def _handle_critical_incident(self, incident_type: str, details: dict):
        """处理紧急事件"""

        # 1. 停止所有Agent
        self.scheduler.emergency_stop_all()

        # 2. 通知创始人
        self._notify_founder(f"🚨 紧急安全事件: {incident_type}\n\n详情: {json.dumps(details, ensure_ascii=False, indent=2)}\n\n系统已自动停止所有Agent。请立即检查。")

        # 3. 创建紧急备份
        backup_manager = BackupManager(self.db)
        backup_path = backup_manager.create_backup("emergency")

        # 4. 根据事件类型采取额外措施
        if incident_type == 'api_key_leaked':
            # 清除API Key
            config = SecureConfig()
            config.set("api_key", "", encrypt=True)
            self._notify_founder("API Key已清除，请重新配置。")

        elif incident_type == 'db_tampered':
            # 提示恢复备份
            self._notify_founder(f"数据库可能被篡改。紧急备份已创建: {backup_path}\n建议立即检查数据完整性。")

    def _handle_high_incident(self, incident_type: str, details: dict):
        """处理高优先级事件"""

        # 1. 暂停相关Agent
        if 'agent_id' in details:
            self.scheduler.pause_agent(details['agent_id'])

        # 2. 通知创始人
        self._notify_founder(f"⚠️ 安全事件: {incident_type}\n\n详情: {json.dumps(details, ensure_ascii=False, indent=2)}\n\n相关Agent已暂停。")

    def _handle_low_incident(self, incident_type: str, details: dict):
        """处理低优先级事件"""

        # 仅记录日志，不中断系统
        logger.warning(f"安全事件(低): {incident_type}, {details}")

    def _notify_founder(self, message: str):
        """通知创始人"""
        # MVP: 通过前端弹窗通知
        # V1.1+: 支持邮件/桌面通知
        ui.notify(message, type='negative', position='top', multi_line=True)
```

```python
class SecuritySelfCheck:
    """安全自检"""

    def __init__(self, db: SecureDatabase, config: SecureConfig):
        self.db = db
        self.config = config

    def run_check(self) -> dict:

        results = {
            "timestamp": datetime.now().isoformat(),
            "checks": [],
            "overall_status": "PASS"
        }

        # 检查1: 审计日志完整性
        audit_query = AuditLogQuery(self.db)
        integrity = audit_query.get_integrity_check()
        results["checks"].append({
            "name": "审计日志完整性",
            "status": "PASS" if integrity['integrity_status'] == 'OK' else "FAIL",
            "details": integrity
        })

        # 检查2: API Key加密
        api_key = self.config.get_encrypted("api_key")
        results["checks"].append({
            "name": "API Key加密存储",
            "status": "PASS" if api_key is not None else "WARN",
            "details": {"configured": api_key is not None}
        })

        # 检查3: 文件权限
        secret_key_path = Path("data/.secret_key")
        if secret_key_path.exists():
            stat = secret_key_path.stat()
            mode = oct(stat.st_mode)[-3:]
            results["checks"].append({
                "name": "密钥文件权限",
                "status": "PASS" if mode == '600' else "WARN",
                "details": {"current_mode": mode, "expected_mode": "600"}
            })

        # 检查4: 数据库备份
        backup_dir = Path("data/backups")
        recent_backups = list(backup_dir.glob("*.db")) if backup_dir.exists() else []
        results["checks"].append({
            "name": "数据库备份",
            "status": "PASS" if len(recent_backups) > 0 else "WARN",
            "details": {"backup_count": len(recent_backups)}
        })

        # 检查5: 沙箱配置
        results["checks"].append({
            "name": "代码沙箱",
            "status": "PASS",
            "details": {"timeout": 30, "max_memory_mb": 512}
        })

        # 汇总
        if any(c['status'] == 'FAIL' for c in results["checks"]):
            results["overall_status"] = "FAIL"
        elif any(c['status'] == 'WARN' for c in results["checks"]):
            results["overall_status"] = "WARN"
```
