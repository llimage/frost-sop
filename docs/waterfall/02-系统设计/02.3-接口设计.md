## 1. 设计目标与约束

### 1.1 设计目标

| 目标 | 说明 |
|------|------|
| G1 | Agent 间通信零延迟（同进程内存通信） |
| G2 | 前端-后端实时同步（≤3 秒延迟） |
| G3 | LLM 调用统一封装（本地/云端透明切换） |
| G4 | 工具调用权限校验 + 审计记录 |
| G5 | 数据访问层抽象（SQLite/ChromaDB 统一接口） |

### 1.2 设计约束

| 约束 | 来源 | 影响 |
|------|------|------|
| 单进程架构 | PRD（内存约束） | Agent 通信用内存队列，非网络 |
| NiceGUI 事件机制 | PRD（前端框架） | 前端-后端绑定用 NiceGUI 原生事件 |
| Ollama API 兼容 OpenAI | 技术选型 | LLM 接口可统一封装 |
| 同步阻塞 I/O | 简化设计 | 单用户无需异步，代码更简单 |
| 创始人 IT 小白 | PRD | 接口错误提示必须中文、友好 |

## 2. Agent 间通信接口

### 2.1 通信模型

MVP 采用**调度器中转模型**：所有 Agent 间通信通过调度器中转，不直接点对点。

```
Agent A ──→ [调度器消息队列] ──→ Agent B
              │
              ├─→ 记录到 audit_log
              ├─→ 广播状态更新到前端
              └─→ 触发成本记录（如调用LLM）
```

**V1.1+ 扩展**: Agent 间直接通信通道、多轮讨论机制。

### 2.2 消息格式

```python
from dataclasses import dataclass
from typing import Optional, Literal
from datetime import datetime
import uuid

@dataclass
class AgentMessage:
    """Agent间通信消息"""
    
    # 基础字段
    msg_id: str = field(default_factory=lambda: str(uuid.uuid4())[:8])
    timestamp: float = field(default_factory=datetime.now().timestamp)
    
    # 路由字段
    from_agent: str                    # 发送者Agent ID
    to_agent: str                      # 接收者Agent ID (或"broadcast"广播)
    
    # 内容字段
    msg_type: Literal[                 # 消息类型
        "request",                     # 请求: 需要对方响应
        "response",                    # 响应: 回复请求
        "notify",                      # 通知: 单向告知
        "advice",                      # 建议: 审计Agent提供建议
        "decision",                    # 决策: 需要创始人确认
        "error"                        # 错误: 执行出错
    ]
    topic: str                         # 消息主题: code_review / task_assign / status_update / ...
    content: str                       # 消息内容
    
    # 关联字段
    task_id: Optional[str] = None     # 关联任务ID
    stage_id: Optional[str] = None    # 关联阶段ID
    
    # 元数据
    metadata: Optional[dict] = None    # 扩展元数据: {cost: 0.5, tokens: 100, model: "qwen2.5:3b"}
    
    def to_dict(self) -> dict:
        """序列化为字典（用于存储和传输）"""
        return {
            "msg_id": self.msg_id,
            "timestamp": self.timestamp,
            "from_agent": self.from_agent,
            "to_agent": self.to_agent,
            "msg_type": self.msg_type,
            "topic": self.topic,
            "content": self.content,
            "task_id": self.task_id,
            "stage_id": self.stage_id,
            "metadata": self.metadata or {}
        }
    
    @classmethod
    def from_dict(cls, data: dict) -> "AgentMessage":
        """从字典反序列化"""
        return cls(**data)
```

### 2.3 通信流程示例

#### 示例 1: 开发 Agent 请求审计 Agent 审查代码

```python
# Step 1: 开发Agent发送审查请求
msg_request = AgentMessage(
    from_agent="agent_dev",
    to_agent="agent_audit",
    msg_type="request",
    topic="code_review",
    content="请审查这段代码:\n```python\ndef query_tasks():\n    conn = sqlite3.connect('data.db')\n    ...\n```",
    task_id="task_001",
    stage_id="task_001_s3",
    metadata={"language": "python", "file_path": "src/db.py"}
)

# Step 2: 调度器接收并转发
scheduler.receive_message(msg_request)
# → 记录audit_log
# → 转发给agent_audit
# → 更新agent_dev状态为"waiting_review"

# Step 3: 审计Agent处理并回复
msg_response = AgentMessage(
    from_agent="agent_audit",
    to_agent="agent_dev",
    msg_type="response",
    topic="code_review_result",
    content="审查结果:\n1. [警告] 未关闭数据库连接，建议用with语句\n2. [建议] 添加参数化查询防止SQL注入\n3. [建议] 添加错误处理try-except",
    task_id="task_001",
    stage_id="task_001_s3",
    metadata={
        "fatal_count": 0,
        "warning_count": 1,
        "suggestion_count": 2,
        "cost": 0.3,
        "tokens": 200
    }
)

# Step 4: 调度器接收回复并转发
scheduler.receive_message(msg_response)
# → 记录audit_log
# → 转发给agent_dev
# → 更新agent_audit状态为"idle"
# → 更新agent_dev状态为"running"
```

### 2.4 调度器通信接口

```python
class MessageBus:
    """Agent间消息总线（调度器内部使用）"""
    
    def __init__(self, db: Database, cost_tracker: CostTracker):
        self.queues: dict[str, Queue] = {}          # AgentID -> Queue
        self.db = db
        self.cost_tracker = cost_tracker
        self.subscribers: dict[str, list[callable]] = {}  # topic -> [callback]
        
    def register_agent(self, agent_id: str):
        """注册Agent，创建专属消息队列"""
        if agent_id not in self.queues:
            self.queues[agent_id] = Queue()
    
    def send_message(self, msg: AgentMessage) -> bool:
        """发送消息到指定Agent"""
        # 1. 记录审计日志
        self.db.audit_log.insert({
            "agent_id": msg.from_agent,
            "operation_type": "agent_comm",
            "target_type": "agent",
            "target_id": msg.to_agent,
            "params": msg.to_dict(),
            "cost": msg.metadata.get("cost", 0),
            "tokens": msg.metadata.get("tokens", 0)
        })
        
        # 2. 记录成本（如有）
        if msg.metadata and "cost" in msg.metadata:
            self.cost_tracker.record_call(
                agent_id=msg.from_agent,
                model=msg.metadata.get("model", "unknown"),
                input_tokens=0,  # Agent通信本身不产生Token
                output_tokens=msg.metadata.get("tokens", 0),
                cost=msg.metadata.get("cost", 0)
            )
        
        # 3. 放入目标Agent队列
        if msg.to_agent == "broadcast":
            # 广播给所有Agent
            for queue in self.queues.values():
                queue.put(msg)
        else:
            if msg.to_agent in self.queues:
                self.queues[msg.to_agent].put(msg)
            else:
                # 目标Agent未注册，记录错误
                self.db.audit_log.insert({
                    "agent_id": "system",
                    "operation_type": "error",
                    "target_type": "agent",
                    "target_id": msg.to_agent,
                    "params": {"error": "Agent not registered", "msg": msg.to_dict()}
                })
                return False
        
        # 4. 触发订阅回调
        if msg.topic in self.subscribers:
            for callback in self.subscribers[msg.topic]:
                try:
                    callback(msg)
                except Exception as e:
                    logger.error(f"订阅回调错误: {e}")
        
        return True
    
    def receive_message(self, agent_id: str, timeout: float = 0.1) -> Optional[AgentMessage]:
        """Agent从自己的队列接收消息（非阻塞）"""
        if agent_id not in self.queues:
            return None
        try:
            return self.queues[agent_id].get(timeout=timeout)
        except Empty:
            return None
    
    def subscribe(self, topic: str, callback: callable):
        """订阅特定主题的消息"""
        if topic not in self.subscribers:
            self.subscribers[topic] = []
        self.subscribers[topic].append(callback)
    
    def get_message_history(self, task_id: str) -> list[AgentMessage]:
        """获取指定任务的所有通信记录"""
        # 从audit_log查询
        records = self.db.audit_log.query(
            operation_type="agent_comm",
            params_filter={"task_id": task_id}
        )
        return [AgentMessage.from_dict(r["params"]) for r in records]
```

## 3. 前端-后端交互接口

### 3.1 NiceGUI 事件模型

NiceGUI 采用**Python 回调函数**模式：前端组件的事件直接绑定到 Python 函数。

```python
from nicegui import ui

# 前端按钮点击 → 直接调用Python函数
def on_send_click():
    user_input = input_field.value
    # 直接调用后端逻辑
    result = scheduler.on_user_input(user_input)
    # 直接更新前端UI
    log_panel.add_log(f"[系统] 任务已创建: {result.task_id}")
    task_result_area.set_content("Agent正在执行中...")

ui.button("发送", on_click=on_send_click)
```

**关键设计**: NiceGUI 的前端-后端是**同进程**的，没有网络 API 层。所有交互通过 Python 函数调用完成。

### 3.2 前端组件-后端数据绑定

```python
class DashboardState:
    """仪表盘共享状态（NiceGUI绑定用）"""
    
    def __init__(self, scheduler: Scheduler):
        self.scheduler = scheduler
        
        # 绑定到UI的响应式数据
        self.agent_status = {}          # {agent_id: {status, progress, status_text}}
        self.task_list = []             # 今日任务列表
        self.log_entries = []           # 日志条目
        self.kpi_data = {}              # KPI数据
        self.current_task_result = None # 当前任务结果
        
        # 启动定时刷新
        ui.timer(3.0, self.refresh_ui)  # 每3秒刷新一次
    
    def refresh_ui(self):
        """定时刷新UI数据"""
        # 从调度器获取最新状态
        self.agent_status = self.scheduler.get_agent_status()
        self.task_list = self.scheduler.get_today_tasks()
        self.kpi_data = self.scheduler.get_kpi_data()
        
        # NiceGUI自动更新绑定的UI组件
        # (具体绑定在UI组件初始化时完成)
    
    def on_user_input(self, text: str):
        """用户输入处理"""
        task = self.scheduler.on_user_input(text)
        self.log_entries.append({
            "time": datetime.now().strftime("%H:%M"),
            "agent": "用户",
            "message": f"下达指令: {text[:50]}..."
        })
        return task
    
    def on_agent_status_change(self, agent_id: str, status: dict):
        """Agent状态变更回调（由调度器触发）"""
        self.agent_status[agent_id] = status
        # NiceGUI自动更新对应卡片
        
    def on_task_complete(self, task_id: str, result: dict):
        """任务完成回调"""
        self.current_task_result = result
        self.log_entries.append({
            "time": datetime.now().strftime("%H:%M"),
            "agent": "系统",
            "message": f"任务 {task_id} 已完成"
        })
```

### 3.4 前端-后端事件接口汇总

| 事件 | 前端触发 | 后端处理 | 前端更新 |
|------|---------|---------|---------|
| 用户输入 | CEO面板输入+发送按钮 | `scheduler.on_user_input()` | 日志面板追加、任务结果区显示spinner |
| Agent状态变更 | 定时轮询(3秒) | `scheduler.get_agent_status()` | Agent卡片状态灯、进度条、成本 |
| 任务完成 | 调度器回调 | `state.on_task_complete()` | 任务结果区显示结果、日志追加 |
| 成本更新 | 每次LLM调用后 | `cost_tracker.record_call()` | KPI卡片刷新 |
| 决策点弹窗 | 主Agent遇到决策点 | `scheduler.pause_for_decision()` | 弹窗显示问题+选项 |
| 用户确认决策 | 弹窗按钮点击 | `scheduler.on_user_confirm()` | 恢复任务执行、关闭弹窗 |
| 能量状态记录 | 能量输入器提交 | `db.energy_log.insert()` | 能量曲线图更新 |
| 日程提醒 | 定时检查(每分钟) | `schedule`表查询 | 桌面通知弹窗 |

## 4. LLM 调用接口

### 4.1 统一 LLM 接口

```python
from dataclasses import dataclass
from typing import Optional, Literal
import requests

@dataclass
class LLMResponse:
    """LLM调用统一响应"""
    content: str                    # 生成的内容
    model: str                      # 实际使用的模型
    provider: str                   # 提供商: ollama/deepseek/openai/kimi
    input_tokens: int               # 输入Token数
    output_tokens: int              # 输出Token数
    cost: float                     # 成本(元)
    latency_ms: int                 # 延迟(毫秒)
    error: Optional[str] = None     # 错误信息(如有)

class LLMClient:
    """统一LLM调用客户端"""
    
    def __init__(self, config: Config, secure_config: SecureConfig, cost_tracker: CostTracker):
        self.config = config
        self.secure_config = secure_config
        self.cost_tracker = cost_tracker
        
        # 本地Ollama客户端
        self.ollama_host = config.get('ollama_host', 'http://localhost:11434')
        self.ollama_model = config.get('ollama_model', 'qwen2.5:3b')
        
        # 云端API客户端（API Key从SecureConfig获取，不在SQLite中存储）
        self.api_key = secure_config.get_encrypted("api_key")
        self.api_base = config.get('api_base_url')
        self.api_model = config.get('api_model')
        self.api_provider = config.get('api_provider', 'deepseek')
        
        # Token单价(元/千Token) - 需要根据实际提供商更新
        self.pricing = {
            'ollama': {'input': 0, 'output': 0},           # 本地模型0成本
            'deepseek': {'input': 0.001, 'output': 0.002},  # 示例价格
            'openai': {'input': 0.003, 'output': 0.006},
            'kimi': {'input': 0.002, 'output': 0.004}
        }
    
    def chat(
        self,
        messages: list[dict],          # OpenAI格式消息列表
        model_preference: Literal['local', 'cloud', 'auto'] = 'auto',
        temperature: float = 0.7,
        max_tokens: Optional[int] = None,
        agent_id: str = 'unknown'
    ) -> LLMResponse:
        """
        统一聊天接口
        
        model_preference:
        - 'local': 强制使用本地Ollama
        - 'cloud': 强制使用云端API
        - 'auto': 根据任务复杂度自动选择(默认)
        """
        
        start_time = time.time()
        
        # 决策: 使用本地还是云端
        use_local = self._decide_model(model_preference, messages)
        
        if use_local:
            response = self._call_ollama(messages, temperature, max_tokens)
        else:
            response = self._call_cloud_api(messages, temperature, max_tokens)
        
        # 记录成本
        latency = int((time.time() - start_time) * 1000)
        response.latency_ms = latency
        
        if not response.error:
            self.cost_tracker.record_call(
                agent_id=agent_id,
                model=response.model,
                provider=response.provider,
                input_tokens=response.input_tokens,
                output_tokens=response.output_tokens,
                cost=response.cost
            )
        
        return response
    
    def _decide_model(
        self,
        preference: Literal['local', 'cloud', 'auto'],
        messages: list[dict]
    ) -> bool:
        """决策使用本地还是云端模型 —— 任务类型驱动路由
        
        路由策略:
        1. 显式偏好优先: local/cloud直接返回
        2. 成本熔断优先: 预算耗尽时强制本地
        3. 任务类型识别: 从system prompt + user content判断任务类型
        4. 长度辅助: 短文本(<1000字)即使复杂任务也可尝试本地
        
        任务类型路由矩阵:
        | 任务类型      | 默认路由 | 理由                  | 熔断fallback |
        |--------------|---------|----------------------|-------------|
        | intent_parse | local   | 短文本分类，3B足够    | local       |
        | code_gen     | cloud   | 代码质量要求高        | local       |
        | code_review  | cloud   | 需发现潜在bug         | local       |
        | summary      | local   | 文本压缩，3B可胜任    | local       |
        | qa           | auto    | 视问题复杂度动态决定  | local       |
        | creative     | cloud   | 创意质量要求高        | local       |
        | data_analysis| local   | 本地有Python辅助      | local       |
        | chat         | local   | 通用对话，3B足够      | local       |
        """
        
        # 1. 显式偏好优先
        if preference == 'local':
            return True
        if preference == 'cloud':
            # 即使选cloud也要检查熔断
            budget_status = self.cost_tracker.check_budget()
            if budget_status == 'cutoff':
                return True  # 熔断时强制本地
            return False
        
        # 2. 成本熔断检查（最高优先级）
        budget_status = self.cost_tracker.check_budget()
        if budget_status == 'cutoff':
            return True  # 熔断状态: 强制本地，即使可能效果差
        
        # 3. 任务类型识别
        task_type = self._detect_task_type(messages)
        
        # 4. 消息总长度（辅助判断）
        total_length = sum(len(m.get('content', '')) for m in messages)
        is_short = total_length < 1000  # 短文本阈值
        is_medium = total_length < 3000  # 中等文本阈值
        
        # 5. 任务类型路由决策
        routing = {
            'intent_parse': True,      # 本地: 短文本分类
            'code_gen': False,         # 云端: 代码质量
            'code_review': False,      # 云端: 审查质量
            'summary': True,           # 本地: 文本压缩
            'qa': None,              # auto: 动态决定
            'creative': False,         # 云端: 创意质量
            'data_analysis': True,   # 本地: Python辅助
            'chat': True,            # 本地: 通用对话
            'unknown': None,         # auto: 无法识别时动态决定
        }
        
        decision = routing.get(task_type, None)
        
        # 6. 动态决策（qa/unknown类型）
        if decision is None:
            # 短文本 → 本地尝试
            if is_short:
                return True
            # 中等文本 + 非代码类 → 本地尝试（失败可fallback）
            if is_medium and task_type not in ('code_gen', 'code_review'):
                return True
            # 长文本或代码类 → 云端
            return False
        
        # 7. 长度覆盖: 即使是cloud任务，如果非常短，尝试本地
        if not decision and is_short and task_type not in ('code_gen', 'code_review'):
            return True
        
        return decision
    
    def _detect_task_type(self, messages: list[dict]) -> str:
        """从消息中识别任务类型
        
        识别策略:
        1. 检查system prompt中的角色定义
        2. 检查user content中的关键词
        3. 返回最可能的任务类型，unknown表示无法识别
        """
        # 合并所有文本用于分析
        all_text = ' '.join([
            m.get('content', '') for m in messages 
            if m.get('role') in ('system', 'user')
        ]).lower()
        
        # 代码生成类（最高优先级，关键词强）
        code_gen_keywords = [
            '生成代码', '写代码', '编写代码', 'code', 'function', 'def ', 'class ',
            '实现', '开发功能', '写个程序', '写一个脚本'
        ]
        if any(kw in all_text for kw in code_gen_keywords):
            # 区分生成和审查
            review_keywords = ['审查', 'review', '检查', 'audit', 'bug', '问题']
            if any(kw in all_text for kw in review_keywords):
                return 'code_review'
            return 'code_gen'
        
        # 意图解析类
        intent_keywords = [
            '意图', 'intent', '分类', '识别', '解析', '判断', '属于哪类',
            '这是什么类型', '请分析需求'
        ]
        if any(kw in all_text for kw in intent_keywords):
            return 'intent_parse'
        
        # 摘要类
        summary_keywords = [
            '摘要', '总结', 'summarize', 'summary', '概括', '提炼', '提取要点',
            '请总结', '总结一下'
        ]
        if any(kw in all_text for kw in summary_keywords):
            return 'summary'
        
        # 数据分析类
        analysis_keywords = [
            '分析数据', '数据分析', '统计', '计算', 'chart', '可视化',
            'pandas', 'dataframe', 'csv', 'excel'
        ]
        if any(kw in all_text for kw in analysis_keywords):
            return 'data_analysis'
        
        # 创意类
        creative_keywords = [
            '创意', '写作', '写文章', '写故事', '文案', '营销', '广告',
            'creative', 'write a', 'draft', 'compose'
        ]
        if any(kw in all_text for kw in creative_keywords):
            return 'creative'
        
        # 问答类（默认fallback）
        qa_keywords = [
            '为什么', '怎么', '如何', '什么是', '解释', '说明', 'help',
            'question', '?', '请问'
        ]
        if any(kw in all_text for kw in qa_keywords):
            return 'qa'
        
        # 无法识别 → 默认chat
        return 'chat'
    
    def _call_ollama(
        self,
        messages: list[dict],
        temperature: float,
        max_tokens: Optional[int]
    ) -> LLMResponse:
        """调用本地Ollama"""
        
        try:
            # Ollama API兼容OpenAI格式
            response = requests.post(
                f"{self.ollama_host}/v1/chat/completions",
                json={
                    "model": self.ollama_model,
                    "messages": messages,
                    "temperature": temperature,
                    "max_tokens": max_tokens or 2048,
                    "stream": False
                },
                timeout=120
            )
            response.raise_for_status()
            data = response.json()
            
            return LLMResponse(
                content=data['choices'][0]['message']['content'],
                model=self.ollama_model,
                provider='ollama',
                input_tokens=data['usage']['prompt_tokens'],
                output_tokens=data['usage']['completion_tokens'],
                cost=0.0,  # 本地模型0成本
                latency_ms=0
            )
            
        except Exception as e:
            return LLMResponse(
                content='',
                model=self.ollama_model,
                provider='ollama',
                input_tokens=0,
                output_tokens=0,
                cost=0.0,
                latency_ms=0,
                error=f"Ollama调用失败: {str(e)}"
            )
    
    def _call_cloud_api(
        self,
        messages: list[dict],
        temperature: float,
        max_tokens: Optional[int]
    ) -> LLMResponse:
        """调用云端API"""
        
        try:
            headers = {
                "Authorization": f"Bearer {self.api_key}",
                "Content-Type": "application/json"
            }
            
            response = requests.post(
                f"{self.api_base}/v1/chat/completions",
                headers=headers,
                json={
                    "model": self.api_model,
                    "messages": messages,
                    "temperature": temperature,
                    "max_tokens": max_tokens or 2048
                },
                timeout=60
            )
            response.raise_for_status()
            data = response.json()
            
            input_tokens = data['usage']['prompt_tokens']
            output_tokens = data['usage']['completion_tokens']
            
            # 计算成本
            pricing = self.pricing.get(self.api_provider, {'input': 0, 'output': 0})
            cost = (input_tokens * pricing['input'] + output_tokens * pricing['output']) / 1000
            
            return LLMResponse(
                content=data['choices'][0]['message']['content'],
                model=self.api_model,
                provider=self.api_provider,
                input_tokens=input_tokens,
                output_tokens=output_tokens,
                cost=cost,
                latency_ms=0
            )
            
        except Exception as e:
            return LLMResponse(
                content='',
                model=self.api_model,
                provider=self.api_provider,
                input_tokens=0,
                output_tokens=0,
                cost=0.0,
                latency_ms=0,
                error=f"云端API调用失败: {str(e)}"
            )
    
    def embed(self, texts: list[str]) -> list[list[float]]:
        """调用Embedding模型(nomic-embed-text via Ollama)"""
        
        try:
            response = requests.post(
                f"{self.ollama_host}/api/embed",
                json={
                    "model": self.config.get('embedding_model', 'nomic-embed-text'),
                    "input": texts
                },
                timeout=30
            )
            response.raise_for_status()
            data = response.json()
            
            return data['embeddings']
            
        except Exception as e:
            logger.error(f"Embedding调用失败: {e}")
            # 返回零向量作为fallback
            return [[0.0] * 768 for _ in texts]
```

from typing import Callable, Any

        调用工具
        
        流程:
        1. 校验权限: Agent是否有权调用此工具
        2. 校验参数: 是否符合params_schema
        3. 执行工具函数
        4. 记录审计日志
        5. 返回结果
        """

            success = False

            "operation_type": "tool_call",
            "target_type": "tool",
            "target_id": tool_name,
            "params": params,
            "result": "SUCCESS" if success else f"FAILED: {error_msg}",
            "duration_ms": duration
        })
        
        # 5. 返回
        return {
            "success": success,
            "result": result,
            "error": error_msg,
            "duration_ms": duration
        }

        """读取文件"""
        # 路径安全检查: 限制在项目目录内
        safe_path = self._sanitize_path(path)
        with open(safe_path, 'r', encoding='utf-8') as f:
            return f.read()
    
    def _file_write(self, path: str, content: str, append: bool = False) -> bool:
        """写入文件"""
        safe_path = self._sanitize_path(path)
        mode = 'a' if append else 'w'
        with open(safe_path, mode, encoding='utf-8') as f:
            f.write(content)
        return True
    
    def _db_query(self, sql: str, params: list = None) -> list:
        """执行查询(仅SELECT)"""
        # 安全检查: 只允许SELECT
        sql_upper = sql.strip().upper()
        if not sql_upper.startswith('SELECT'):
            raise ToolError("仅允许SELECT查询")
        
        return self.db.query(sql, params or [])
    
    def _code_execute(self, code: str, timeout: int = 30) -> dict:

        from security import CodeSandbox  # 从02.5安全设计导入
        sandbox = CodeSandbox(timeout=timeout)
        return sandbox.execute(code)
    
    def _git_commit(self, message: str, files: list = None) -> bool:
        """Git提交"""
        import subprocess
        if files:
            subprocess.run(['git', 'add'] + files, check=True)
        else:
            subprocess.run(['git', 'add', '-A'], check=True)
        subprocess.run(['git', 'commit', '-m', message], check=True)
        return True
    
    def _sanitize_path(self, path: str) -> Path:
        """路径安全检查: 限制在项目目录内"""
        base = Path.cwd()
        target = (base / path).resolve()
        
        # 确保目标路径在项目目录内
        if not str(target).startswith(str(base)):
            raise ToolError(f"路径越界: {path}")
        
        # 禁止访问敏感目录
        forbidden = ['.git', '.env', 'data/solo_ops.db', 'data/chromadb']
        for f in forbidden:
            if f in str(target):
                raise ToolError(f"禁止访问: {path}")
        
        return target

import sqlite3
from pathlib import Path
from typing import Optional, list, dict, Any

class Database:
    """SQLite数据库访问层"""
    
    def __init__(self, db_path: str = "data/solo_ops.db"):
        self.db_path = Path(db_path)
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        self._connection: Optional[sqlite3.Connection] = None
    
    def _get_connection(self) -> sqlite3.Connection:
        """获取数据库连接(单连接复用)"""
        if self._connection is None:
            self._connection = sqlite3.connect(self.db_path)
            self._connection.row_factory = sqlite3.Row  # 返回字典形式结果
            self._connection.execute("PRAGMA foreign_keys = ON")
        return self._connection
    
    def query(self, sql: str, params: list = None) -> list[dict]:
        """执行查询(SELECT)"""
        conn = self._get_connection()
        cursor = conn.execute(sql, params or [])
        return [dict(row) for row in cursor.fetchall()]
    
    def execute(self, sql: str, params: list = None) -> int:
        """执行写入(INSERT/UPDATE/DELETE)"""
        conn = self._get_connection()
        cursor = conn.execute(sql, params or [])
        conn.commit()
        return cursor.lastrowid
    
    def transaction(self):
        """事务上下文管理器"""
        return self._get_connection()
    
    def close(self):
        """关闭连接"""
        if self._connection:
            self._connection.close()
            self._connection = None

        """根据ID获取任务"""
        results = self.db.query("SELECT * FROM tasks WHERE task_id = ?", [task_id])
        return results[0] if results else None
    
    def get_today_tasks(self) -> list[dict]:
        """获取今日任务"""
        return self.db.query("""
            SELECT * FROM tasks 
            WHERE date(created_at) = date('now')
            ORDER BY 
                CASE priority 
                    WHEN 'P0' THEN 1 WHEN 'P1' THEN 2 
                    WHEN 'P2' THEN 3 WHEN 'P3' THEN 4 WHEN 'P4' THEN 5 
                END,
                created_at DESC
        """)
    
    def update_status(self, task_id: str, status: str, result_summary: str = None):
        """更新任务状态"""
        sql = "UPDATE tasks SET status = ?"
        params = [status]
        if result_summary:
            sql += ", result_summary = ?"
            params.append(result_summary)
        if status == 'completed':
            sql += ", completed_at = datetime('now')"
        elif status == 'running':
            sql += ", started_at = datetime('now')"
        sql += " WHERE task_id = ?"
        params.append(task_id)
        
        self.db.execute(sql, params)

class AuditLogRepository:

            INSERT INTO audit_log

             params, result, duration_ms, cost, tokens)
            VALUES (datetime('now'), ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """
        self.db.execute(sql, [
            record.get('agent_id'), record.get('operation_type'),
            record.get('target_type'), record.get('target_id'),
            json.dumps(record.get('params', {})),
            record.get('result'), record.get('duration_ms', 0),
            record.get('cost', 0.0), record.get('tokens', 0)
        ])
    
    def query(self, operation_type: str = None, params_filter: dict = None, 
              limit: int = 50) -> list[dict]:
        """查询审计记录"""
        sql = "SELECT * FROM audit_log WHERE 1=1"
        params = []
        
        if operation_type:
            sql += " AND operation_type = ?"
            params.append(operation_type)
        
        # params_filter简化处理: 仅支持JSON字段的模糊匹配
        if params_filter:
            for key, value in params_filter.items():
                sql += f" AND params LIKE ?"
                params.append(f'%"{key}": "{value}"%')
        
        sql += " ORDER BY timestamp DESC LIMIT ?"
        params.append(limit)
        
        return self.db.query(sql, params)

class StructuredLogger:
    """结构化日志: 输出JSON格式，便于分析"""
    
    def __init__(self, name: str):
        self.logger = logging.getLogger(name)
    
    def info(self, event: str, **kwargs):
        """记录信息日志"""
        self.logger.info(f"[{event}] {json.dumps(kwargs, ensure_ascii=False)}")
    
    def error(self, event: str, error: Exception = None, **kwargs):
        """记录错误日志"""
        data = {"event": event, **kwargs}
        if error:
            data["error_type"] = type(error).__name__
            data["error_message"] = str(error)
        self.logger.error(json.dumps(data, ensure_ascii=False))
    
    def agent_action(self, agent_id: str, action: str, **kwargs):
        """记录Agent操作"""
        self.info("agent_action", agent_id=agent_id, action=action, **kwargs)
    
    def llm_call(self, agent_id: str, model: str, tokens: int, cost: float, **kwargs):
        """记录LLM调用"""
        self.info("llm_call", agent_id=agent_id, model=model, 
                  tokens=tokens, cost=cost, **kwargs)
    
    def tool_call(self, agent_id: str, tool: str, success: bool, duration_ms: int, **kwargs):
